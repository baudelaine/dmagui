SQL queries for relations

SQL syntax
SELECT ... as FK_NAME, ... as PK_NAME, ... as FKTABLE_NAME, ... as PKTABLE_NAME, ... as KEY_SEQ, ... as FKCOLUMN_NAME, ... as PKCOLUMN_NAME FROM .. = ?
About parameter(s)
? stand for any table.
e.g.
SELECT fk as FK_NAME, pk as PK_NAME, fkTable as FKTABLE_NAME, pkTable as PKTABLE_NAME, seq as KEY_SEQ, fkColumn as FKCOLUMN_NAME, pkColumn as PKCOLUMN_NAME FROM relationsTable WHERE fkTable = ?

Everything in bold is mandatory.
e.g. MAXIMO
SELECT 'FK_' || TARGETOBJ || '_' || OBJECTNAME || '_' || LOGICALRELATIONSHIP.LOGICALRELATIONSHIPID as FK_NAME, 'PK_' || OBJECTNAME || '_' || TARGETOBJ || '_' || LOGICALRELATIONSHIP.LOGICALRELATIONSHIPID as PK_NAME, TARGETOBJ as FKTABLE_NAME, OBJECTNAME as PKTABLE_NAME, KEYNUM as KEY_SEQ, TARGETKEY as FKCOLUMN_NAME, PARENTKEY as PKCOLUMN_NAME FROM LOGICALRELATIONSHIP join LRKEYS on LOGICALRELATIONSHIP.LOGICALRELATIONSHIPID = LRKEYS.LOGICALRELATIONSHIPID WHERE TARGETOBJ = ?

SQL queries for labels and descriptions

SQL syntax
SELECT ... AS Table_Name, ... AS Table_Label FROM ... IN (?)
About parameter(s)
(?) stand for a list of tables.
e.g.
SELECT table AS Table_Name, label AS Table_Label FROM tablelabels where table IN (?) and lang = 'fr'

Everything in bold is mandatory.
e.g. MAXIMO
SELECT MAXIMO.MAXTABLE.TABLENAME AS Table_Name, MAXIMO.MAXOBJECT.DESCRIPTION AS Table_Label FROM MAXIMO.MAXTABLE join MAXIMO.MAXOBJECT on MAXIMO.MAXTABLE.TABLENAME = MAXIMO.MAXOBJECT.OBJECTNAME where MAXTABLE.TABLENAME IN (?)
e.g. LV
SELECT systable.tableid AS Table_Name, singular AS Table_Label FROM systable join sdc on sdc.tableid = systable.tableid where systable.tableid IN (?)
SQL syntax
SELECT ... AS Table_Name, ... AS Table_Description FROM ... IN (?)
About parameter(s)
(?) stand for a list of tables.
e.g.
SELECT table AS Table_Name, description AS Table_Description FROM tablelabels where table IN (?) and lang = 'fr'

Everything in bold is mandatory.
e.g. MAXIMO
SELECT MAXIMO.MAXTABLE.TABLENAME AS Table_Name, MAXIMO.MAXOBJECT.DESCRIPTION AS Table_Description FROM MAXIMO.MAXTABLE join MAXIMO.MAXOBJECT on MAXIMO.MAXTABLE.TABLENAME = MAXIMO.MAXOBJECT.OBJECTNAME where MAXTABLE.TABLENAME IN (?)
e.g. LV
SELECT systable.tableid AS Table_Name, tabledoc AS Table_Description FROM systable join sdc on sdc.tableid = systable.tableid where systable.tableid IN (?)
SQL syntax
SELECT ... AS Table_Name, ... AS Column_Name, ... AS Column_Label FROM ... = ? ... IN (?)
About parameter(s)
? stand for any table.
(?) stand for all fields of a table.
e.g.
SELECT table AS Table_Name, column AS Column_Name, label AS Column_Label FROM columnlabels where table = ? and column IN (?) and lang = 'fr'

Everything in bold is mandatory.
e.g. MAXIMO
SELECT maxattribute.objectname AS Table_Name, maxattribute.attributeName AS Column_Name, maxattribute.title AS Column_Label FROM maxattribute where maxattribute.objectname = ? and maxattribute.attributeName IN (?)
e.g. LV
SELECT tableid AS Table_Name, columnid AS Column_Name, columnlabel AS Column_Label FROM syscolumn where tableid = ? and columnid IN (?)
